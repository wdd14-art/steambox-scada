    // 1. Cek Komunikasi & Maintenance
    if (commOp === 0) return "UNIT TIDAK DIPAKAI";

    if (maintMode) return "MODE MAINTENANCE (KONTROL MANUAL)";
    if (tempRaw >= 3000) return "ERROR SENSOR (OPENLOOP/HHHH)";
    if (statusSelesai === 1) return "PROSES SELESAI - SILAKAN KOSONGKAN TANGKI";
    // 2. Mode Preheat
    if (modePreheat === 1) {
        if (runStop === 0) return "SEDANG PEMANASAN";
        else if (totalDetikPreheat === 0) return "SIAP PEMANASAN - SILAKAN TEKAN START";
        else return "PRE-HEAT DIHENTIKAN (PAUSED)";
    } 
    // 3. Mode Cooking
    else {
        if (runStop === 1) {
            if (hasRecipe) return (sisaDetik === 0) ? "RESEP TERPASANG - SILAKAN TEKAN START" : "MESIN BERHENTI (PAUSED)";
            else return (totalDetikPreheat > 0) ? "PEMANASAN SELESAI - STEAMBOX SIAP UNTUK PEMASAKAN" : "STEAMBOX KOSONG";
        } else {
            if (tempRaw < 1000) return "MENUNGGU MENDIDIH (< 100 C)";
            else return "SEDANG MEMASAK (MENDIDIH)";
        }
    }
}
